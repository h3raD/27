int find_element(double* array, int n, double element)
{
	for (int i = 0; i < n; i++) {
		if (element == array[i])
			return i;
	}
	return -1;
}

double* full_elements(double* array, int n)
{
	for (int i = 0; i < n; i++) {
		array[i] = -30 + rand() % 61;
	}
}

int put_elements(double* array, int n)
{
	for (int i = 0; i < n; i++) {
		printf("%lf\n", array[i]);
	}
}
double* calc_elements(double* array, int n)
{
	for (int i = 0; i < n; i++) {
		if (array[i] < 0) {
			array[i] = 0; 
		}
	}
}
int sum_elements(double* array, int begin, int end)
{
	int sum = 0;

	while (begin <= end) {
		sum += array[begin];
		begin += 1;
	}
	return sum;
}
int poiskpol1(double* array, int n)
{
	int poslpol;
	for (int i = 0; i < n; i++) {
		if (array[i] > 0) {
			poslpol = array[i];
		}
	}
	return poslpol;
}
